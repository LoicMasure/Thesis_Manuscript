\newglossaryentry{cryptography}{
	name={cryptography},
	description={The art of designing \gls{encryption} and \gls{decryption} systems}
}
\newglossaryentry{primitive}{
	name={cryptographic primitive},
	description={Algorithm that is the building block of a crypto-system, \eg{} processing \gls{encryption}, \gls{decryption}, or hach operations}
}
\newglossaryentry{encryption}{
	name={encryption},
	description={Transforming a plaintext message into a ciphertext meaningless for anyone whose the message is not addressed}
}
\newglossaryentry{decryption}{
	name={decryption},
	description={Transforming back a ciphertext message into the initial plaintext message}
}
\newglossaryentry{cryptanalysis}{
	name={cryptanalysis},
	description={The art of guessing the plaintext message behind a ciphertext message}
}
\newglossaryentry{cryptology}{
	name={cryptology},
	description={The union of \gls{cryptography} and \gls{cryptanalysis}}
}
\newglossaryentry{sym_crypto}{
	name={symmetric cryptography},
	description={Field of cryptography where the encryption key also serves for decryption}
}
\newglossaryentry{asym_crypto}{
	name={asymmetric cryptography},
	description={Field of cryptography where the encryption key differs from the decryption one}
}
\newglossaryentry{block_cipher}{
	name={block cipher},
	description={\gls{sym_crypto} algorithm that splits a plaintexts message into blocks of fixed size that are then encrypted and decrypted separately}
}
\newglossaryentry{diffusion}{
	name={diffusion},
	description={Method to dissipate into long range statistics -- \ie{} into statistical structure involving long combinations of letters in the ciphertext~\cite{shannon_communication_1949}}
}
\newglossaryentry{confusion}{
	name={confusion},
	description={Method making the relation between the simple statistics of ciphertexts and the simple description of the secret key a very ocmplex and involved one~\cite{shannon_communication_1949}}
}
\newglossaryentry{linAttack}{
	name={linear cryptanalysis},
	description={A class of \gls{cryptanalysis} based on linear approximations of the target \gls{primitive}}
}
\newglossaryentry{diffAttack}{
	name={differential cryptanalysis},
	description={A class of \gls{cryptanalysis} based on the study of how slight variations in some input messages span variations in the cipher text, depending on the nature of the secret key}
}
\newglossaryentry{sigmaAlg}{
	name={\(\sigma\)-algebra},
	description={A collection of subsets of a global set that is closed under complement, and is closed under \gls{countable} unions; \emph{Tribu} in French}
}
\newglossaryentry{countable}{
	name={countable},
	description={Which is in bijection with \(\natSet\)}
}
\newglossaryentry{finite}{
	name={finite},
	description={Which is in bijection with \(\llbracket1,N\rrbracket\). \(N\) is called the \emph{cardinality} of the set}
}
\newglossaryentry{gmorph}{
	name={group morphism},
	description={A map that preserves the mathematical group structure between two spaces}
}
\newglossaryentry{vecSpace}{
	name={vector space},
	description={A linear algebraic structure that is stable by linear combinations where scalars are taken from a field~\cite[Chap. 7.1]{terras_2018}}
}
\newglossaryentry{prodScal}{
	name={scalar product},
	description={A bilinear mapping to \(\realSet_+\) that is symmetric and semi-positive}
}
\newglossaryentry{partDeriv}{
	name={partial derivative},
	description={derivative of a function with respect to only one variable}
}
\newglossaryentry{gradient}{
	name={{g}radient},
	description={The vector made of every \gls{partDeriv} of a function}
}
\newglossaryentry{jacob}{
	name={Jacobian matrix},
	plural={Jacobian matrices},
	description={The matrix whose rows are the transposed gradient of each elementary function \(\xxx \mapsto f(\xxx)[i] \in \realSet\)}
}
\newglossaryentry{commut}{
	name={commutative},
	description={Binary operation \(\star\) such that \(a \star b = b \star a\)}
}
\newglossaryentry{assoc}{
	name={associative},
	description={Binary operation \(\star\) such that \((a \star b) \star c = a \star (b \star c)\)}
}
\newglossaryentry{d&c}{
	name={divide-and-conquer},
	description={A way to solve a problem by dividing it into several sub-problems that are simpler to solve}
}
\newglossaryentry{support}{
	name={support},
	description={Set where a function is non zero}
}
\newglossaryentry{sigProc}{
	name={signal processing},
	description={Mathematical field that studies the techniques of analysis and processing of signals}
}
\newglossaryentry{stochastic}{
	name={stochastic},
	description={Which involves randomness}
}
\newglossaryentry{consistent}{
	name={consistent},
	description={Finite proxy approach (\eg{} estimation, optimization) that converges to the target result}
}
\newglossaryentry{characteristic}{
	name={characteristic},
	description={Least integer \(p\) such that \(p \cdot 1_{\gf{}{}} = 0_{\gf{}{}}\) in a field \(\gf{}{}\)}
}
\newglossaryentry{cyclic}{
	name={cyclic},
	description={Finite group such that there is one element spanning all the others, by successive application of the inner law \(\gplaw\)}
}
\newglossaryentry{convex}{
	name={convex},
	description={Function whose epigraph is convex. See~\cite{boyd_convex_2004}}
}
\newglossaryentry{critical}{
	name={critical},
	description={Point where the gradient is zero}
}
\newglossaryentry{voisinnage}{
	name={neighbourhood},
	description={Set of points that are close to a point for a given norm}
}
\newglossaryentry{vertical}{
	name={vertical},
	description={Attacks involving univariate leakages}
}
\newglossaryentry{ae}{
	name={almost everywhere},
	description={Which is only not true on a set of measure zero}
}
\newglossaryentry{nph}{
	name={NP-hard},
	description={A problem for which there is no algorithm providing a solution in a polynomial time}
}
\newglossaryentry{hp}{
	name={hyper-parameter},
	description={A parameter which cannot be optimized using an optimization algorithm, \eg{} discrete parameters of the architecture}
}
\newglossaryentry{resnet}{
	name={Resnet},
	description={Deep Residual Neural Networks, equipped with \emph{skip} connections in the architecture~\cite{he_deep_2015}}
}
\newglossaryentry{sparse}{

}
